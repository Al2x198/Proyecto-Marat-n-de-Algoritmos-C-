#include <iostream>
#include <string>
#include <cctype>
using namespace std;

int main() {
    string palabra;
    int contador = 0;

    cout << "Ingrese una palabra: ";
    cin >> palabra;

    for (char c : palabra) {
        c = tolower(c);  // Convertir a min√∫scula
        if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u') {
            contador++;
        }
    }

    cout << "La palabra contiene " << contador << " vocal(es)." << endl;

    return 0;
}
