#include <iostream>
using namespace std;

int main() {
    int i = 0;
    int limite;

    cout << "Ingresa el limite: ";
    cin >> limite;

    cout << "\nContando de 15 en 15:\n";

    while (i <= limite) {
        cout << i << " ";
        i += 15;
    }

    cout << endl;
    return 0;
}
